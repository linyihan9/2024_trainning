```
class Restaurant:
    def __init__(self,name,type):
        self.name = name
        self.type = type
    def describe_res(self):
        print(self.name+" "+self.type)
    def open(self):
        print("Open")

if __name__ == '__main__':
    res = Restaurant("a","b")
    res.describe_res()
    res.open()
```
